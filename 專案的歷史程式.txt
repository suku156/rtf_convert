//用來將處理好的陣列轉成.wmf圖檔
void saveAsWMF(const std::vector<unsigned char>& bytes,int& convertCount){
   
    // 設定檔案名稱
    std::string filename;
    filename = "image"  + std::to_string(convertCount) + ".wmf";
     
    // 設定寫入檔案
    std::ofstream out(filename,std::ios::binary);
    if(!out.is_open()){ // 防止檔案無法開啟
        std::wcerr<<L"錯誤 : 無法建立輸出檔案!\n";
        return;
    }

    // 此行的目的是 將位元組資料（unsigned char array）強制轉為文字指標（char*）
    out.write(reinterpret_cast<const char*>(bytes.data()), bytes.size());
    out.close();

    std::wcout<<L"圖片轉換成功"<<std::endl;
}    

//用來將處理好的十六進位字串轉為 byte陣列的函式
void hexToBytes(const std::string& pictBlock,int& convertCount ){
    std::vector<unsigned char> bytes;
    std::size_t len = pictBlock.length();

    //簡易檢查程序判斷整體長度是否為偶數
    if((len % 2) != 0 ){
        std::wcerr << "警告 字串長度非偶數,將忽略最後一位數";
        len--;
    }

    //進行轉換
    for (size_t i = 0; i < len; i += 2) {
        std::string byteString = pictBlock.substr(i, 2);
        if (std::isxdigit(byteString[0]) && std::isxdigit(byteString[1])) { //確認存入的兩個字元是否為十六進制
            try { // 嘗試進行下列動作
                unsigned char byte = static_cast<unsigned char>(std::stoul(byteString, nullptr, 16));
                bytes.push_back(byte);
            } catch (const std::exception& e) { // 捕捉所有意外錯誤
                std::wcerr << L"錯誤：無法解析 "<< std::endl; 
            }
        } else {
            std::wcerr << L"警告：跳過非十六進位片段 "<<std::endl;
        }
    }
    
    // 將陣列轉存成 .wmf檔案
    saveAsWMF(bytes,convertCount);

}

//用來處理 WMF 圖檔用
void procesWmf(std::string pictBlock,int convertCount){
    //用來移除不需要的第一行
    std::size_t first_newline_pos = pictBlock.find('\n');
    if(first_newline_pos != std::string::npos){
        pictBlock.erase(0,first_newline_pos+1);
    }

    //用來移除尾部的殘留大括號
    pictBlock.pop_back();

    //用來去除資料中的空白與換行
    pictBlock.erase(std::remove_if(pictBlock.begin(),pictBlock.end(),[](unsigned char c){
        return std::isspace(c);
    }),pictBlock.end());

    //用來將處理好的十六進位字串轉為 byte陣列
    hexToBytes(pictBlock,convertCount);

}

    }

// 用來清理 夾在控制符中間的各種大括號{},並保留可見字
  static size_t stripPropertyGroup(std::string& C , size_t pos){
    if( pos >= C.size() || C[pos] != '{') return pos; // 簡易防呆
    size_t i = pos + 1;
    int depth = 1;
    size_t groupStart = pos;

    // 此段用來找到對應的大括號的下半部的位置(允許巢狀結構)
    while(i<C.size() && depth >0){
      if(C[i] == '{'){
        ++depth;
      }else if(C[i] == '}'){
        --depth;;
      }
      ++i;
    }
    size_t groupEnd = i; // 找到 整個大括號的末尾後記錄下來

    // 取出找到的群組範圍做 移除控制符 與 大括號 的任務
    std::string inner = C.substr(groupStart + 1 , (groupEnd - groupStart) - 2); //順手處理掉最外層的{}
    
    // 把剩餘範圍內的大括號都轉為空白
    for(char& ch : inner){
      if(ch == '{' || ch == '}'){
        ch = ' ';
      }
    }

    // 此段用來次一些藏在正常字串中的控制符 例如:c1,c2 等等
    std::string out;
    out.reserve(inner.size());
    for(size_t x = 0; x < inner.size();){
      // 檢查是否是 c + 數字 並且左右格都是空格
      if(inner[x] == ' '){
        if(inner[x+1] == 'c'){
          size_t y = x + 1;
          while( y < inner.size() && (inner[y] >= '0' && inner[y] <= '9')){ // 假設後面數字會有兩個以上
            y++;
          }
          if(inner[y] == ' ' && y + 1 <inner.size()){
            y++;
          }
          x = y;
        }
      }else{
       out.push_back(inner[x++]); // 不是控制符的就直接加入字串 , 這裡有用到 ++ 在變數後方的特性 
      }
    }
    inner.swap(out); // 使用 swap 安全又高效率

    // 把清理後的字串寫回原字串中
    C.replace(groupStart,groupEnd-groupStart,inner);
    //
    return inner.size(); 
  }

// 判斷是否為大寫或小寫的英文
  static inline bool isAlpha(char c){
    return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z');
  }
  static inline bool isAlpha(unsigned char c){ // 無負號版本
    return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z');
  }
  // 判斷是否為數字或負號
  static inline bool isNumOrSign(char c){
    return (c >= '0' && c<= '9') || c == '-';
  }
  static inline bool isAsciiDigit(unsigned char c){ // 純數字版本,unsigned 不會有負號
    return (c >= '0' && c<= '9');
  }
  
  // 用來 判斷是否真的是 RTF 控制群組的開頭 "{\" 
  static inline bool isControlGroupStart(const std::string& s,size_t pos){
    return pos + 1 < s.size() && s[pos] == '{' && s[pos+1] == '\\';
  }
  
  // 用來跳過 rtf 中標準控制符
  static size_t skipControlWord(const std::string& s,size_t i){
    // 先檢查開頭是不是控制符的反斜線
    size_t n = s.size();
    if(i >= n || s[i] != '\\') return i; 
    
    ++i; //確認是控制符就往後一格 
    
    if(i < n && isAlpha(static_cast<unsigned char>(s[i]))){
      //控制字
      while(i < n && isAlpha(static_cast<unsigned char>(s[i]))) ++i;
      
      //參數 (可為負號+數字)
      if(i < n && (s[i] == '-' || isAsciiDigit(static_cast<unsigned char>(s[i])))){
        ++i;
        while(i < n && isAsciiDigit(static_cast<unsigned char>(s[i]))) ++i;
      }
      
      //控制符的尾部有時候會有空格也要檢查有就跳過
      if(i < n && s[i] == ' ') ++i;
      return i;
    }else{ // 例外 通常是一些符號 例如 \{, \}, \\, \~, \-, \_ 等
      if(i < n) ++i;
      return i;
    }
  }
  
  // 找到與 左大括號對應的右大括號結尾
  static size_t findGroupEnd(const std::string& s,size_t start){
    size_t n = s.size();
    if(start >= n || s[start] != '{') return std::string::npos;
    int depth = 1;
    size_t i = start+1;
    
    while(i < n && depth > 0){
      if(s[i] == '{'){
        if(isControlGroupStart(s,i))++depth;
        ++i;
      }else if(s[i] == '}'){
        --depth;
        ++i;
      }else if(s[i] == '\\'){
        size_t newi = skipControlWord(s,i);
        if(newi == i){
          ++i; // 如果沒有跳過就只加一
        }else{
          i = newi; // 有就更新成跳過的
        }
      }else{
        ++i;
      }
    }
  
  return (depth == 0)? i : std::string::npos;
}

    }

    // 使用 lambda 建立一個判斷是否具有 關鍵字的 控制群組開頭 
    auto shouldDiscardWhole = [&](const std::string& w,size_t bracePos) -> bool{
      if (w == "fonttbl" || w == "colortbl" || w == "stylesheet" ||
        w == "info"    || w == "generator" || w == "filetbl" ||
        w == "listtbl" || w == "themedata" || w == "object" ||
        w == "pict") {
      return true;
      }
      // 特別控制符 {\*\...} 關鍵字可忽略
      if(bracePos + 2 <= n && text[bracePos +2] == '*') return true;
      return false;
    };

  //清理{\loch 開頭的群組
  void cleanlochGroup(std::string& Cleaned){
    size_t pos = 0;
    while((pos = Cleaned.find("{\\loch",pos)) != std::string::npos){
      size_t end = Cleaned.find('}',pos);
      if(end == std::string::npos) break;

      std::string group = Cleaned.substr(pos,end-pos+1);

      //最小防呆,如果使用沒有先轉譯導致誤刪會記錄進 log 當中以當作提醒
      if(group.find("\\u") != std::string::npos){
        std::cerr << "[Warning] Detected \\uXXXX sequence in style group. "
                  << "Please run decodeUnicodeAndClean() before text cleanup.\n";
      }

      //清除控制符,保留正文
      std::string cleaned;
      bool inCtrl = false;
      for (size_t i = 0; i < group.size(); ++i) {
        char c = group[i];
        if (c == '\\') {
          inCtrl = true;  // 進入控制符區
          continue;
        } 
        if (inCtrl) {
          // 控制符結束：遇到空白或非英數
          if (!std::isalpha(static_cast<unsigned char>(c)) &&
            !std::isdigit(static_cast<unsigned char>(c))) {
            inCtrl = false;
          }
          continue;
        }  
        if (c != '{' && c != '}') {
            cleaned.push_back(c);
        }
      }

      //如果清出來的字串的開頭或結尾有多餘的換行或空白就清除它
      while(!cleaned.empty() && (cleaned.front() == '\n' || cleaned.front() == '\r' || cleaned.front() == ' ')){
        cleaned.erase(cleaned.begin());//檢查字串第一個字是換行或空白就清除它,持續到不是
      }
      while(!cleaned.empty() && (cleaned.back() == '\n' || cleaned.back() == '\r' || cleaned.back() == ' ')){
        cleaned.pop_back();//檢查字串最後一個字是換行或空白就清除它,持續到不是
      }

      //進行替換動作
      if (!cleaned.empty()) {
        Cleaned.replace(pos, end - pos + 1, cleaned);
        pos += cleaned.size();
      } else { //修改空字串邏輯,如果原群組沒有需要的正文就整個群組刪掉
        Cleaned.erase(pos,end-pos+1);
      }
    }
  }

  //清理{\hich 開頭的群組
  void cleanhichGroup(std::string& content){
    size_t pos = 0;
    while((pos = content.find("{\\hich",pos)) != std::string::npos){
      size_t end = content.find('}',pos);
      if(end == std::string::npos) break;

      //取出開頭與結尾範圍內的文字進行判斷與加工
      std::string group = content.substr(pos,end - pos + 1);

      //判斷擷取出的段落中有無 \uXXXX 這樣的控制符
      bool hasUnicode = group.find("\\u") != std::string::npos;//
      if(hasUnicode){ //如果有找到控制符,就跳過這個群組去找下一個目標
        pos = end + 1;
        continue; 
      }

      //去除群組中的控制符
      std::string cleaned;
      bool inCtrl = false;
      for(size_t i = 0;i<group.size();i++){
        char c = group[i];
        if(c == '\\'){
          inCtrl = true;
        }else if(inCtrl){ //滿足條件才判斷是非控制符,在記錄到字串中
          if(!std::isalpha(static_cast<unsigned char>(c)) && !std::isdigit(static_cast<unsigned char>(c))){
            inCtrl = false;
          }
        }else if(c != '{' && c != '}'){ // 大括號也不要
          cleaned.push_back(c);
        }
      }

      //如果清出來的字串的開頭或結尾有多餘的換行或空白就清除它
      while(!cleaned.empty() && (cleaned.front() == '\n' || cleaned.front() == '\r' || cleaned.front() == ' ')){
        cleaned.erase(cleaned.begin());//檢查字串第一個字是換行或空白就清除它,持續到不是
      }
      while(!cleaned.empty() && (cleaned.back() == '\n' || cleaned.back() == '\r' || cleaned.back() == ' ')){
        cleaned.pop_back();//檢查字串最後一個字是換行或空白就清除它,持續到不是
      }

      if(!cleaned.empty() && std::isdigit(static_cast<unsigned char>(cleaned.front()))){ 
        char c = '\x01'; // 開頭是數字加入特別符號避免誤判
        cleaned.insert(cleaned.begin(),c);
      }

      // 確認條件都正確就進行替換的動作
      if(!cleaned.empty()){
        content.replace(pos,end - pos +1,cleaned);
        pos += cleaned.size();
      }else{
        content.erase(pos,end-pos+1);
      }
    }
  }

    // 如果不是檔案開頭部份 , 在中間部份出現具有關鍵字的群組就進行刪除動作
    if(pos > 512 && shouldDiscardWhole(headWord , pos)){
      size_t end = findGroupEnd(text,pos);
      if(end != std::string::npos){
        text.erase(pos,end -pos);
        return pos;
      }else{
        return pos + 1;
      }
    }

    // 如過是非常見控制群組就進行額外處理
    std::string buffer;
    int depth = 1; // 用 int 避免有不對稱的 大括號數量
    while(i < n && depth >0){
      char ch = text[i];

      if(ch == '{'){ // 遇到左大括號要特別判斷是 巢狀迴圈開頭 還是 另一個 控制群組開頭
        if(isControlGroupStart(text,i)){
          // 是巢狀控制群組,就遞迴抽取處理
          i = stripPropertyGroup(text,i);
          continue;
        }else{
          ++depth;
          ++i;
        }
      }else if(ch == '}'){
        --depth;
        ++i;
        if(depth == 0) break; // 遇到左大括號就檢查是否已經平衡了
      }else if(ch == '\\'){
        size_t n2 = text.size();
        if(i + 1<n2 && isAlpha(static_cast<unsigned char>(text[i+1]))){
          size_t j = i+2;
          while(j < n2 && isAlpha(static_cast<unsigned char>(text[j]))) ++j;
          std::string w = text.substr(i+1,j - (i+1)); // 擷取不含反斜線的英文部份

          //數字.負號以及空白部份處理
          if(j < n2 && (text[j] == '-' || isAsciiDigit(static_cast<unsigned char>(text[j])))){
            ++j;
            while(j < n2 && isAsciiDigit(static_cast<unsigned char>(text[j]))) ++j;
          }
          if(j < n2 && text[j] == ' ')++j;

          //換行與空白的特別處裡
          if(w == "par" || w == "line"){
            buffer.push_back('\n');
          }else if(w == "tab"){
            buffer.push_back('\t');
          }

          i = j;
        }else { //如果反斜線後方式是其餘控制符號
          if(i + 1 < n){
            char sym = text[i+1];
            switch(sym){ // 依照是什麼特殊符號,決定要加什麼進字串
              case '{': buffer.push_back('{'); i += 2; break;
              case '}': buffer.push_back('}'); i += 2; break;
              case '\\': buffer.push_back('\\'); i += 2; break;
              case '~': buffer.push_back(' '); i += 2; break;
              case '-': i += 2; break;
              case '_': buffer.push_back('-'); i += 2; break;
              default: i += 2; break; 
            }
          }else{ // 非指定特別符號就忽略
            ++i;
          } 
        }
      }else{ // 非控制符的可見文字就全部加入字串
        buffer.push_back(ch);
        ++i;
      }
    }

    // 用處理的結果來取代原本的群組
    size_t end = i;
    text.erase(pos,end - pos);
    text.insert(pos,buffer);
    return pos + buffer.size();
  }
  

//用來清理 pard段落
  void textPardClean(std::string& Cleaned){
    const std::string keyPard = "\\pard";
    size_t pos = Cleaned.find(keyPard);

    while(pos != std::string::npos){ // 在找不到開頭前都持續進行
      size_t i = pos;
      
      // 把 \pard 後面一長串控制符刪掉
      consumeControlWords(Cleaned,i);

      // 上方會持續到找到 { 為止,這裡就繼續處理大括號, 因為可能是連續群組因此用迴圈 
      while(Cleaned[i] == '{' && i < Cleaned.size()){
        i = stripPropertyGroup(Cleaned , i); 
      }
      
      // 處理完一段後,繼續找下一個開頭
      pos = Cleaned.find(keyPard,i);
    }
  }

//用來將string(utf-8) 轉換成 wstring(utf-16)
std::wstring stringToWstring(const std::string& str) {
    std::wstring wstr(str.size(), L'\0');
    std::mbstate_t state{};
    const char* src = str.data();
    wchar_t* dst = &wstr[0];
    std::use_facet<std::codecvt<wchar_t, char, std::mbstate_t>>(std::locale()).in(
        state, src, src + str.size(), src, dst, dst + wstr.size(), dst);
    wstr.resize(dst - &wstr[0]);
    return wstr;
}

//用來判斷 utf 體系的空格與換行等
  size_t utfSpaceBytes(const std::string& s,size_t pos){
    auto B = [&](size_t i){return static_cast<unsigned char>(s[i]);};
    if(pos >= s.size()) return 0;

    unsigned char c = B(pos);

    // ASCII 區塊的判斷
    if(c <= 0x7F){
      if(c == ' ' || c == '\t' || c == '\n' || c == '\r' || c == '\v' || c == '\f') return 1;
      return 0;
    }

    // UTF 體系區域判斷
    if (pos + 1 < s.size() && B(pos) == 0xC2 && B(pos+1) == 0xA0) return 2; // NBSP
    if (pos + 2 < s.size()) {
      if (B(pos)==0xE3 && B(pos+1)==0x80 && B(pos+2)==0x80) return 3; // 全形空白
      if (B(pos)==0xEF && B(pos+1)==0xBB && B(pos+2)==0xBF) return 3; // BOM / 零寬不換行
      if (B(pos)==0xE2 && B(pos+1)==0x80) {
        unsigned char t = B(pos+2);
        if ((t >= 0x80 && t <= 0x8A) || // 各種 EN/EM space
             t == 0x8B || t == 0x8C || t == 0x8D || // 零寬空白
             t == 0xA8 || t == 0xA9)
        { // 行分隔/段分隔
          return 3;
        } 
               
      }
    }

    return 0; 
  }
  

// 小工具以此來判斷 utf佔據幾 byte
  inline size_t utf8CharLen(unsigned char c){
    if(c < 0x80){
      return 1;  // 第一個位元組小於 0x80,代表這是標準ASCII
    }else if((c >> 5) == 0x6){
      return 2; // 代表 2-byte utf8 延伸字元（歐洲字母、部分注音)會用到
    }else if((c >> 4) == 0xE){
      return 3; // 代表 3-byte UTF-8 幾乎所有中文、日文、韓文字都是這類型
    }else if((c >> 3) == 0x1E){
      return 4; // 4-byte UTF-8 高碼位符號（Emoji、罕見文字）
    }else{
      return 1; // 如果沒查出來假設其是ASCII
    }
  }
  
  //用來移除所有控制群組
  void eraseControlsGroup(std::string& Cleand){
    const std::string target = "{\\rtf";
    size_t headerPos = Cleand.find(target);
    if(headerPos == std::string::npos) return;

    size_t pos = headerPos + target.size();
    while((pos = Cleand.find("{\\",pos)) != std::string::npos){
      
      // 加入高位元判斷保護機制
      unsigned char c = static_cast<unsigned char>(Cleand[pos]);

      // 偵測到非 ASCII 範圍的 byte 就跳過整個字元
      if(c >= 0x80){
        pos += utf8CharLen(c);//小工具會給要跳過幾個字元
        continue;
      }
      
      
      // 特別留下的圖片標記群組要將其保留
      if(Cleand.compare(pos,12,"{\\*\\imgblock") == 0){
        size_t end = pos + 1;
        int depth = 1;
        while(end < Cleand.size() && depth > 0){
          unsigned char endc = static_cast<unsigned char>(Cleand[end]);
          if(endc < 0x80){ // 只對ASCII 編碼範圍內的括號做判斷
            if(Cleand[end] == '{'){
              depth++;
            }else if(Cleand[end] == '}'){
              depth--;
            }
            end++;
          }else{ // 不是 ASCII 範圍內的就跳過整個字元
            end += utf8CharLen(endc);
          }
        }
        pos = end ; //整個要保留的群組直接跳過
        continue;
      }
      
      
      // 一般控制群組找到就刪除
      size_t end = pos +1;
      int depth = 1;
      while(end <Cleand.size() && depth > 0){
        unsigned char endc = static_cast<unsigned char>(Cleand[end]);
        if(endc < 0x80){
          if(Cleand[end] == '{'){
            depth++;
          }else if(Cleand[end] == '}'){
            depth--;
          }
          end++;
        }else{
          end += utf8CharLen(endc);
        }
      }

      if(depth != 0){
        std::wcerr << L"rtf檔案群組括號不平衡,導致無法順利清除"<<std::endl;
        return;
      }else{
        Cleand.erase(pos,end-pos);
      }
    }
  }

  //進行一些換行符的轉換
  void SymbolConversion(std::string& Cleaned){
    const std::map<std::string,std::string> symbolMap = {
      {"\\par", "\n"},
      {"\\line", "\n"},
      {"\\tab", "\t"},
      {"\\emdash", "—"},
      {"\\endash", "-"},
      {"\\bullet", "•"}
    };

    for(const auto& [pattern,replacement]:symbolMap){
      size_t pos = 0;
      while((pos = Cleaned.find(pattern,pos)) != std::string::npos){
        Cleaned.replace(pos,pattern.size(),replacement);
        pos += replacement.size(); 
      }
    }

    //合併連續換行
    std::string result;
    result.reserve(Cleaned.size());
    bool lastWasNewline = false;

    for(char c: Cleaned){
      if(c == '\n'){
        if(!lastWasNewline){
          result.push_back(c);
          lastWasNewline = true;
        }
      }else{
        result.push_back(c);
        lastWasNewline = false;
      }
    }
    
    //把處理好的交換
    Cleaned.swap(result);
  }

  //最後剩下的的控制碼清理
  void finalRTFccClean(std::string& Cleand){
    std::string result;
    result.reserve(Cleand.size());

    for(size_t i = 0;i < Cleand.size();){
      
      //此段用來保留 圖片標記群組 {\*\imgblock[[IMG_0]]}
      if (i + 12 < Cleand.size() && Cleand.compare(i, 12, "{\\*\\imgblock") == 0) {
        size_t end = i + 1;
        int depth = 1;
        while (end < Cleand.size() && depth > 0) {
          if (Cleand[end] == '{') depth++;
          else if (Cleand[end] == '}') depth--;
          end++;
        }
        // 若括號不平衡就提前結束
        if (depth != 0) {
          std::wcerr << L"[警告] imgblock 群組括號不平衡，結尾未閉合。\n";
          end = Cleand.size();
        }
        // 原封不動保留整段
        result.append(Cleand.substr(i, end - i));
        i = end;
        continue;
      }
      
      //此段用來跳過一般控制符
      if(Cleand[i] == '\\'){
        size_t j = i + 1;
        while(j < Cleand.size() && std::isalpha(static_cast<unsigned char>(Cleand[j]))) ++j;
        while(j < Cleand.size() && std::isdigit(static_cast<unsigned char>(Cleand[j]))) ++j;
        if (j < Cleand.size() && Cleand[j] == ' ') ++j;
        i = j;
        continue;
      }

      //保留剩餘的所有文字
      result.push_back(Cleand[i]);
      ++i;
    }

    //加入移除文章前方空白的處理
    while(!result.empty() && (result.front() == '\n' || result.front()=='\r')){
      result.erase(result.begin());
    }

    Cleand.swap(result);
  }

  //移除結尾多餘的大括號
  void trimEndBraces(std::string& Cleand){
    if(Cleand.empty()) return;
    
    size_t end = Cleand.size();
    size_t maxCheck = (end < 5)? end : 5;
    size_t removeCount = 0;
    
    // 由尾端往前檢查最多5個字元
    for (size_t i = 0; i < maxCheck; ++i) {
      if(Cleand[end - 1 - i] == '}'){
        ++removeCount;
      }else break; // 一旦遇到不是 '}' 就停止
    }

    if(removeCount > 0){
      Cleand.erase(end-removeCount);
    }
  }

  //先移除 pard 避免誤判換行符
  void onlyCleanPard(std::string& Cleaned){
    std::string result;
    result.reserve(Cleaned.size());

    for(size_t i =0;i < Cleaned.size();){
      unsigned char c = static_cast<unsigned char>(Cleaned[i]);

      //非ASCII 範圍內的字就直接保留
      if(c >= 0x80){
        size_t len = utf8CharLen(c);
        result.append(Cleaned,i,len);
        i += len;
        continue;
      }

      // 檢查是否是目標控制符 \pard
      if(Cleaned[i] == '\\' && i+5 < Cleaned.size() && Cleaned.compare(i,5,"\\pard") == 0){
        i += 5; // 是目標控制符就整組跳過不保留
        continue;
      }

      //非目標都直接保留
      result.push_back(Cleaned[i]);
      ++i;
    }

    Cleaned.swap(result);
  }

// 用來處理 可繼續執行類錯誤 
class ErrorLogger : public IErrorHandler{
  FileContext* ctx_ = nullptr;
  std::ofstream logFile_;
  bool logCreated_ = false;
   
  void createLogFile(){
    if(!ctx_) return;
    std::filesystem::path logPath = ctx_->getOutputDir() / "error_log.txt";
    
    logFile_.open(logPath,std::ios::app);
    if(logFile_.is_open()){
      logCreated_ = true;
      logFile_ << "==== Error Log Start ====\n";
    }

  }
public:
  
  void setContext(FileContext* ctx){
    ctx_ = ctx;
  }

  void reportError(const std::wstring& message) override{
    std::wcerr << L"[內部資訊處理錯誤]"<< message << L"\n";
  }
 
  void logerror(const std::string& content){
    if(!logCreated_){
      createLogFile();
    }
    if(logFile_.is_open()){
      logFile_ << content << "\n";
      logFile_.flush();
    }
  }
};

// 用來建立資料夾並且加上安全檢查
bool ensureDirectory(const std::filesystem::path& dir){
  std::error_code ec;
  
  //1.如果已經有檔案,先檢查原本路徑使否真的有指向物件
  if(std::filesystem::exists(dir,ec)){
    if(!ec && std::filesystem::is_directory(dir,ec)){
      return true; // 如果該物件是資料夾就 OK
    }else{
      return false; // 不是就是檔案 回傳 否
    }
  }
  
  //2.如果沒東西就嘗試建立資料夾
  std::filesystem::create_directories(dir,ec);
  if(ec){ // 建立失敗回傳 否
    return false;  
  }

  return std::filesystem::is_directory(dir,ec);
}


bool hasRTFVersion(std::ifstream& input){
    input.clear();
    std::streampos oldPos = input.tellg();
    input.seekg(0,std::ios::beg);

    unsigned char buf[6] = {0};
    input.read(reinterpret_cast<char*>(buf),6);
    std::streamsize n = input.gcount();

    if(n < 6){
      input.clear();
      input.seekg(oldPos);
      return false;
    }

    if(buf[0] == '{' && buf[1] == '\\' && buf[2] == 'r' &&
       buf[3] == 't' && buf[4] == 'f'){
      
      bool ok = std::isdigit(static_cast<unsigned char>(buf[5]));
      input.clear();
      input.seekg(oldPos);
      return ok;
    }
    
    input.clear();
    input.seekg(oldPos);
    return false;
  }

class DocumentBuilder{
public:
  Document build(const std::string& content){
    Document doc;
    Paragraph* currentPar;
    std::string result;
    
    currentPar = &doc.addParagraph();// 第一段

    for(size_t i =0;i<content.size();i++){
      if(content[i]=='[' && i + 6 <= content.size() && content.compare(i,6,"[[IMG_") == 0){
        size_t j = i + 6;
        if(j>= content.size() || !isdigit(static_cast<unsigned char>(content[j]))){
          // 非正確圖片當作普通文字
          result.push_back(content[i]);
          continue;
        }
        
        size_t numstart = j;// 紀錄數字開頭
        while(j<content.size() && isdigit(static_cast<unsigned char>(content[j]))) j++;// 找到最後一個數字
        
        // 確認結尾符合標記符結尾,就執行紀錄
        if(j+1 < content.size() && content[j] == ']' && content[j+1] == ']'){
          if(!result.empty()){
            currentPar -> addNode(std::make_unique<TextNode>(result));
            result.clear();
            
          }
          
          std::string numResult = content.substr(numstart,j - numstart);
          currentPar -> addNode(std::make_unique<ImageNode>(numResult));
         
          i = j+1;
          continue;
        }else{ // 預防非正確標記符結尾
          // 當作普通文字
          result.push_back(content[i]);
          continue;
        }

      }else if(content[i] == '\n' && i+1 < content.size() && content[i+1] == '\n'){
        if(!result.empty()){
          currentPar -> addNode(std::make_unique<TextNode>(result));
          result.clear();
        }
        currentPar = &doc.addParagraph();
        i++;// 跳過空行迴圈會在 +1
        continue;
      }else{
        result.push_back(content[i]);
      }
    }
    
    // 最後收尾
    if(!result.empty()){
      currentPar -> addNode(std::make_unique<TextNode>(result));
    }

    return doc;
  }
};

// 圖片類別用的錯誤偵測
class pictErrorDetector{
  
  size_t findHexStart(std::string_view s){
    size_t i = 0;
    const size_t n = s.size();

    auto isHexDigit = [](unsigned char c){
      return std::isxdigit(c) != 0;
    };

    while(i < n){
      unsigned char c = static_cast<unsigned char>(s[i]);

      if(c == '\\'){
        i++;
        if(i >= n) break;

        unsigned char next = static_cast<unsigned char>(s[i]);

        if(std::isalpha(next)){
          while(i<n && std::isalpha(static_cast<unsigned char>(s[i]))) ++i;
          while(i<n && std::isdigit(static_cast<unsigned char>(s[i]))) ++i;
          continue;
        }else{
          i++;
          continue;
        }
      }else if(isHexDigit(c)){
        return i;
      }else{
        ++i;
      }
    }

    return std::string::npos;
  }
  
  inline std::vector<uint8_t> convertWmfHeader(std::string_view content,size_t start){
    std::vector<uint8_t> header;

    size_t maxChars = 36;
    for(size_t i = start;i < start + maxChars; i+= 2){
      if(i+1 >= content.size()) break;

      char h = content[i];
      char l = content[i+1];

      uint8_t value = (std::stoi(std::string(1,h),nullptr,16) << 4) |
                       std::stoi(std::string(1,l),nullptr,16);

      header.push_back(value);
    }

    return header;
  }
  
  static uint16_t readWord(const std::vector<uint8_t>& data, size_t offset) {
    return static_cast<uint16_t>(data[offset] | (data[offset + 1] << 8));
  } 

  static uint32_t readDword(const std::vector<uint8_t>& data, size_t offset) {
    return static_cast<uint32_t>(data[offset] |
                                (data[offset + 1] << 8) |
                                (data[offset + 2] << 16) |
                                (data[offset + 3] << 24));
  }
  
  bool checkBraceBalance(std::string_view content){
    int depth = 0;
    for(char p: content){
      if(p == '{') depth++;
      else if(p == '}') depth--;
      if(depth < 0){
        return false;
      }
    }
    if(depth != 0){
      return false;
    }
    
    return true;
  }

  bool checkDataContent(std::string_view content){
    size_t start = findHexStart(content);
    if(start == std::string::npos) return false;
    size_t end = content.rfind('}');
    if(end == std::string::npos || start >= end) return false;

    size_t hexCount = 0; // 檢查hex字元數量是否為偶數 
    for(size_t i =start; i < end ; i++){ // end 為右大括號 刻意避開所以 < end
      unsigned char uc = static_cast<unsigned char>(content[i]);
      
      if(std::isxdigit(uc)){
        hexCount++;
      }else if(content[i] == ' ' || content[i] == '\n' ||
               content[i] == '\r'|| content[i] == '\t' || 
               content[i] == '\v')
      {
        continue;
      }else{
        return false;
      }
    }
    
    if(hexCount % 2 != 0){
      return false;
    }
    
    std::vector<uint8_t> header = convertWmfHeader(content,start);
    uint32_t fileSize     = readDword(header, 6); 
    
    size_t declaredBytes = fileSize * 2;
    
    size_t actualBytes = hexCount / 2;
    
    if(declaredBytes != actualBytes) return false;

    return true;
  }

  bool checkWmfHeader(std::string_view content){
    // 1) 找到 hex stream 的開始位置
    size_t start = findHexStart(content);
    if (start == std::string::npos)
        return false;

    // 2) 將 hex 轉成二進位 bytes
    std::vector<uint8_t> data = convertWmfHeader(content, start);
    if (data.size() < 18) return false;

    

    // 3) 檢測 Placeable WMF Header (Aldus Header)
    size_t offset = 0;
    if (data.size() >= 22) {
      uint32_t key = (uint32_t)data[0] | ((uint32_t)data[1] << 8) |
                     ((uint32_t)data[2] << 16) | ((uint32_t)data[3] << 24);
      if (key == 0x9AC6CDD7) {
        // 跳過 22 bytes，真正標準 WMF Header 由此開始
        offset = 22;
      }
    }

    // 4) 確保還有足夠空間放標準 WMF Header（18 bytes）
    if (data.size() < offset + 18) return false;

    

    // 5) 解析 Standard WMF Header
    uint16_t type       = readWord(data, offset + 0);
    uint16_t headerSize = readWord(data, offset + 2);
    uint16_t version    = readWord(data, offset + 4);
    uint32_t fileSize   = readDword(data, offset + 6);   // (word count)
    uint16_t numObjects = readWord(data, offset + 10);
    uint32_t maxRecord  = readDword(data, offset + 12);
    uint16_t numParams  = readWord(data, offset + 16);

    // 6) 基礎合法性檢查
    if (type != 1 && type != 2) return false;
    
    if (headerSize != 9) return false; // WMF headerSize = 9 words = 18 bytes
    
    if (version != 0x0300 && version != 0x0100) return false;
    
    if (numObjects > 30000) return false;
    
    if (numParams  > 10) return false;
    
    if (maxRecord == 0) return false;
    
    // 7) 基本合理性：fileSize（WORD）至少要 >= header
    if (fileSize < headerSize) return false;
    
    return true;
  }
  
  public:
  ErrorSystem::ErrorInfo detect(std::string_view content) {
    ErrorSystem::ErrorInfo info;
    
    // 結構檢查
    if(!checkBraceBalance(content)){
      info.add(ErrorSystem::ErrorType::Picture_Structural_BraceMismatch,
               ErrorSystem::ErrorLevel::Recoverable,
               ErrorSystem::ErrorCategory::Picture,
               L"[圖片區塊錯誤] 大括號數量不平衡");
    }
    
    // 內容檢查
    if(!checkDataContent(content)){
      info.add(ErrorSystem::ErrorType::Picture_Content_DataError,
               ErrorSystem::ErrorLevel::Recoverable,
               ErrorSystem::ErrorCategory::Picture,
               L"[圖片區塊錯誤] 圖片內容區塊錯誤");
    }

    // 圖檔是否為轉換目標格式檢查
    if(!checkWmfHeader(content)){
      info.add(ErrorSystem::ErrorType::Picture_Content_FormatError,
               ErrorSystem::ErrorLevel::Recoverable,
               ErrorSystem::ErrorCategory::Picture,
               L"[圖片區塊錯誤] 圖片目標格式錯誤");
    }
    

    return info;
  }
};

ErrorSystem::ErrorInfo detectBig5(const std::string& content){
    
    ErrorSystem::ErrorInfo info;
    
    const size_t strEnd = content.size();
    
    size_t i = 0;
    while(i < strEnd){
      unsigned char c = static_cast<unsigned char>(content[i]);

      // ASCII 區(合法直接跳過)
      if(c < 0x80){
        i++;
        continue;
      }

      // 非法的開頭 byte 範圍
      if(c < 0x81 || c > 0xFE){
        info.add(ErrorSystem::ErrorType::Decode_ANSI_InvalidByte,
                 ErrorSystem::ErrorLevel::Recoverable,
                 ErrorSystem::ErrorCategory::Decode,
                 L"[ANSI_Big5 錯誤偵測區錯誤]: 非法的 byte 開頭");
        i++;
        continue;
      }
      std::wcout<<L"有通過 第一次的判斷式\n";
      // 檢查有無第二個byte (半字偵測)
      if(i + 1 >= strEnd){
        info.add(ErrorSystem::ErrorType::Decode_ANSI_InvalidByte,

// utf16 轉碼前專用的錯誤偵測
class Utf16ErrorDetector : public IErrorDetector{
public:
  ErrorSystem::ErrorInfo detect(const std::string& content) override{
    ErrorSystem::ErrorInfo info;
    std::vector<uint8_t> bytes(content.begin(),content.end());
    
    // 防止檔案過小
    if(bytes.size() < 2){
      info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
               ErrorSystem::ErrorLevel::Recoverable,
               ErrorSystem::ErrorCategory::Decode,
               L"[UTF16 轉碼前錯誤偵測錯誤]: 檔案長度不足 ,缺少BOM");
      return info;
    }
    
    //檢查 BOM
    bool littleEndian;
    if(!detectBOM(bytes,littleEndian)){
      info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
               ErrorSystem::ErrorLevel::Recoverable,
               ErrorSystem::ErrorCategory::Decode,
               L"[UTF16 轉碼前錯誤偵測錯誤]: 找不到合法的 BOM 開頭");
      return info;
    }

    // 檢查長度(UTF16 必定是 2 的倍數)
    checkStreamLength(bytes,info);

    // 檢查整個 utf16結構是否合法
    checkSurrogatePairs(bytes,littleEndian,info);

    return info;
  }

private:
  
  bool detectBOM(const std::vector<uint8_t>& bytes,bool& littleEndian){
    if(bytes[0] == 0xFF && bytes[1] == 0xFE){
      littleEndian = true;
      return true;
    }
    if(bytes[0] == 0xFE && bytes[1] == 0xFF){
      littleEndian = false;
      return true;
    }
    return false;
  }

  void checkStreamLength(const std::vector<uint8_t>& bytes , ErrorSystem::ErrorInfo& info){
    size_t dataBytes = bytes.size() - 2;

    if(dataBytes %2 != 0){
      info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
               ErrorSystem::ErrorLevel::FatalLocal,
               ErrorSystem::ErrorCategory::Decode,
               L"[UTF16 轉碼前錯誤偵測錯誤]: UTF16 長度不是 2 的倍數");
    }
  }

  uint16_t readCodeUnit(const std::vector<uint8_t>& bytes,size_t pos, bool littleEndian){
    return littleEndian ? (bytes[pos] | (bytes[pos+1] << 8)) :
                          ((bytes[pos] << 8) | bytes[pos+1]);
  }
  
  bool isHighSurrogate(uint16_t cu){
    return (cu >= 0xD800 && cu <= 0xDBFF);
  }

  bool isLowSurrogate(uint16_t cu){
    return (cu >= 0xDC00 && cu <= 0xDFFF);
  }

  bool isNonCharacter(uint32_t cp){
    return (cp >= 0xFDD0 && cp <= 0xFDEF) ||
           (cp == 0xFFFE || cp == 0xFFFF);
  }
  
  void checkSurrogatePairs(const std::vector<uint8_t>& bytes,
                           bool littleEndian , ErrorSystem::ErrorInfo& info)
  {
    for(size_t i = 2; i+1 < bytes.size(); i += 2){
      uint16_t cu = readCodeUnit(bytes,i,littleEndian);

      // 低位檢查 低位代理不可單獨出現
      if(isLowSurrogate(cu)){
        info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
                 ErrorSystem::ErrorLevel::Recoverable,
                 ErrorSystem::ErrorCategory::Decode,
                 L"[UTF16 轉碼前錯誤偵測錯誤]: 孤立的低位代理(low surrogate)");
        continue;
      }
      
      // 高位檢查 需跟低位代理正確配對
      if(isHighSurrogate(cu)){
         
        if(i+3 >= bytes.size()){
          info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
                   ErrorSystem::ErrorLevel::Recoverable,
                   ErrorSystem::ErrorCategory::Decode,
                   L"[UTF16 轉碼前錯誤偵測錯誤]: 高位代理後資料不足,缺少低位代理");
          continue;
        }

        uint16_t low = readCodeUnit(bytes,i+2,littleEndian);

        if(!isLowSurrogate(low)){
          info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
                   ErrorSystem::ErrorLevel::Recoverable,
                   ErrorSystem::ErrorCategory::Decode,
                   L"[UTF16 轉碼前錯誤偵測錯誤]: 高位代理後不是合法低位代理");
          continue;
        }

        uint32_t cp = 0x10000 + ((cu - 0xD800) << 10) + (low - 0xDC00);
        if(isNonCharacter(cp)){
          info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
                   ErrorSystem::ErrorLevel::Recoverable,
                   ErrorSystem::ErrorCategory::Decode,
                   L"[UTF16 轉碼前錯誤偵測錯誤]: 代理對應值錯誤");
        }

        i += 2;
        continue;
      }

      if(isNonCharacter(cu)){
        info.add(ErrorSystem::ErrorType::Decode_UTF16_ByteError,
                 ErrorSystem::ErrorLevel::Recoverable,
                 ErrorSystem::ErrorCategory::Decode,
                 L"[UTF16 轉碼前錯誤偵測錯誤]: 出現 non-character (如 FFFF)");
      }
    }
  }
};

class IEncodingConverter{
public:
  virtual ~IEncodingConverter() = default;
  virtual std::string convert(const std::string& rtfContent) = 0;
};

// 將 utf16 轉成 utf8 的轉碼類別
class Utf16ToUtf8Converter : public IEncodingConverter{
  void appendUTF8(std::string& out, uint32_t codePoint) {
    if (codePoint <= 0x7F) {
      out.push_back(static_cast<char>(codePoint));
    } else if (codePoint <= 0x7FF) {
      out.push_back(static_cast<char>(0xC0 | (codePoint >> 6)));
      out.push_back(static_cast<char>(0x80 | (codePoint & 0x3F)));
    } else if (codePoint <= 0xFFFF) {
      out.push_back(static_cast<char>(0xE0 | (codePoint >> 12)));
      out.push_back(static_cast<char>(0x80 | ((codePoint >> 6) & 0x3F)));
       out.push_back(static_cast<char>(0x80 | (codePoint & 0x3F)));
    } else {
      out.push_back(static_cast<char>(0xF0 | (codePoint >> 18)));
      out.push_back(static_cast<char>(0x80 | ((codePoint >> 12) & 0x3F)));
      out.push_back(static_cast<char>(0x80 | ((codePoint >> 6) & 0x3F)));
      out.push_back(static_cast<char>(0x80 | (codePoint & 0x3F)));
    }
  }
public:
  std::string convert(const std::string& rtfContent) override{
    std::vector<uint8_t> buf(rtfContent.begin(),rtfContent.end());
    std::string out;

    
    bool littleEndian = true;

    
    if (buf.size() >= 2) {
      if (buf[0] == 0xFF && buf[1] == 0xFE) littleEndian = true;
      else if (buf[0] == 0xFE && buf[1] == 0xFF) littleEndian = false;
    }

    // 轉碼從 index 2 開始
    for (size_t i = 2; i + 1 < buf.size(); i += 2) {

      // 組成 UTF-16 codeUnit
      uint16_t cu = littleEndian
        ? (buf[i] | (buf[i+1] << 8))
        : ((buf[i] << 8) | buf[i+1]);

      // 不檢查 → 直接組 surrogate pair
      if (cu >= 0xD800 && cu <= 0xDBFF && i + 3 < buf.size()) {
        uint16_t low = littleEndian
          ? (buf[i+2] | (buf[i+3] << 8))
          : ((buf[i+2] << 8) | buf[i+3]);

        uint32_t cp = 0x10000 + (((cu - 0xD800) << 10) | (low - 0xDC00));
        appendUTF8(out, cp);
        i += 2;   // consume both
      }
      else {
        // 一般 BMP 直接轉
        appendUTF8(out, cu);
      }
    }

    return out;
  }
};

size_t findGroupEnd(const std::string& text,size_t groupStart) const{
    size_t braceCount = 1;
    size_t pos = groupStart + 1;
    bool inTheHex = false;
    size_t hexCount = 0;
    auto isHexDigit = [](char c){
     return (c >= '0' &&  c <= '9') ||
            (c >= 'a' &&  c <= 'f') ||
            (c >= 'A' &&  c <= 'F'); 
    };

    auto isHexWhitespace = [](char c){
      return c == ' ' || c == '\t' || c == '\r' || c == '\n';
    };

    while(pos < text.size() && braceCount > 0){
      
      if(inTheHex && text[pos] == '\\'){ // 在進入連續 hex 區域後, 碰到控制符就視為結尾大括號缺失  
        return std::string::npos;
      }
      
      if(text[pos] == '{'){
        braceCount++;
      }else if(text[pos] == '}'){
        braceCount--;
      }

      if(isHexDigit(text[pos])){
        hexCount++;
      }else if(isHexWhitespace(text[pos])){
        // 遇到空格或換行 不影響連續 hex 判斷
      }else{
        hexCount = 0;
        inTheHex = false;
      }

      if(hexCount >= 32){
        inTheHex = true;
      }

      pos++;
    }

    if(braceCount != 0){
      return std::string::npos;
    }
    

    return pos;
  }
